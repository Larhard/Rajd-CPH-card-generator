\newcommand{\setbackground}[1]{
    \backgroundsetup{%
        scale=1,
        color=black,
        opacity=1,
        angle=0,
        contents={%
            \includegraphics[width=\paperwidth,height=\paperheight]{#1}
        }%
    }%
}

\newcommand{\setrajdfont}{
    \fontspec[
        Path          = \currfileabsdir,
        Ligatures     = TeX,
        UprightFont   = fonts/CasablancaAntique.ttf,
    ]{CasablancaAntique}
}
